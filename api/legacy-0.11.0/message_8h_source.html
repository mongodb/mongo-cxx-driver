188 // concat multiple buffers - noop if <2 buffers already, otherwise can be expensive copy
189 // can get rid of this if we make response handling smarter
190 void concat() {
191 if ( _buf || empty() ) {
192 return;
193 }
194
195 verify( _freeIt );
196 int totalSize = 0;
197 for (std::vector< std::pair< char *, int > >::const_iterator i = _data.begin();
198 i != _data.end(); ++i) {
199 totalSize += i->second;
200 }
201 char *buf = (char*)malloc( totalSize );
202 char *p = buf;
203 for (std::vector< std::pair< char *, int > >::const_iterator i = _data.begin();
204 i != _data.end(); ++i) {
205 memcpy( p, i->first, i->second );
206 p += i->second;
207 }
208 reset();
209 _setData( (MsgData*)buf, true );
210 }
