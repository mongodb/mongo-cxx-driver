29 this large function is in header to facilitate inline-only use of bson
30 */
31 inline int compareElementValues(const BSONElement& l, const BSONElement& r) {
32 int f;
33
34 switch ( l.type() ) {
35 case EOO:
36 case Undefined: // EOO and Undefined are same canonicalType
37 case jstNULL:
38 case MaxKey:
39 case MinKey:
40 f = l.canonicalType() - r.canonicalType();
41 if ( f<0 ) return -1;
42 return f==0 ? 0 : 1;
43 case Bool:
44 return *l.value() - *r.value();
45 case Timestamp:
46 // unsigned compare for timestamps - note they are not really dates but (ordinal + time_t)
47 if ( l.date() < r.date() )
48 return -1;
49 return l.date() == r.date() ? 0 : 1;
50 case Date:
51 {
52 long long a = (long long) l.Date().millis;
53 long long b = (long long) r.Date().millis;
54 if( a < b )
55 return -1;
56 return a == b ? 0 : 1;
57 }
58 case NumberLong:
59 if( r.type() == NumberLong ) {
60 long long L = l._numberLong();
61 long long R = r._numberLong();
62 if( L < R ) return -1;
63 if( L == R ) return 0;
64 return 1;
65 }
66 goto dodouble;
67 case NumberInt:
68 if( r.type() == NumberInt ) {
69 int L = l._numberInt();
70 int R = r._numberInt();
71 if( L < R ) return -1;
72 return L == R ? 0 : 1;
73 }
74 // else fall through
75 case NumberDouble:
76dodouble:
77 {
78 double left = l.number();
79 double right = r.number();
80 if( left < right )
81 return -1;
82 if( left == right )
83 return 0;
84 if( isNaN(left) )
85 return isNaN(right) ? 0 : -1;
86 return 1;
87 }
88 case jstOID:
89 return memcmp(l.value(), r.value(), 12);
90 case Code:
91 case Symbol:
92 case String:
93 /* todo: a utf sort order version one day... */
94 {
95 // we use memcmp as we allow zeros in UTF8 strings
96 int lsz = l.valuestrsize();
97 int rsz = r.valuestrsize();
98 int common = std::min(lsz, rsz);
99 int res = memcmp(l.valuestr(), r.valuestr(), common);
100 if( res )
101 return res;
102 // longer string is the greater one
103 return lsz-rsz;
104 }
105 case Object:
106 case Array:
107 return l.embeddedObject().woCompare( r.embeddedObject() );
108 case DBRef: {
109 int lsz = l.valuesize();
110 int rsz = r.valuesize();
111 if ( lsz - rsz != 0 ) return lsz - rsz;
112 return memcmp(l.value(), r.value(), lsz);
113 }
114 case BinData: {
115 int lsz = l.objsize(); // our bin data size in bytes, not including the subtype byte
116 int rsz = r.objsize();
117 if ( lsz - rsz != 0 ) return lsz - rsz;
118 return memcmp(l.value()+4, r.value()+4, lsz+1 /*+1 for subtype byte*/);
119 }
120 case RegEx: {
121 int c = strcmp(l.regex(), r.regex());
122 if ( c )
123 return c;
124 return strcmp(l.regexFlags(), r.regexFlags());
125 }
126 case CodeWScope : {
127 f = l.canonicalType() - r.canonicalType();
128 if ( f )
129 return f;
130 f = strcmp( l.codeWScopeCode() , r.codeWScopeCode() );
131 if ( f )
132 return f;
133 f = strcmp( l.codeWScopeScopeDataUnsafe() , r.codeWScopeScopeDataUnsafe() );
134 if ( f )
135 return f;
136 return 0;
137 }
138 default:
139 verify( false);
140 }
141 return -1;
142 }

144 /* wo = "well ordered"
145 note: (mongodb related) : this can only change in behavior when index version # changes
146 */
147 inline int BSONElement::woCompare( const BSONElement &e,
148 bool considerFieldName ) const {
149 int lt = (int) canonicalType();
150 int rt = (int) e.canonicalType();
151 int x = lt - rt;
152 if( x != 0 && (!isNumber() || !e.isNumber()) )
153 return x;
154 if ( considerFieldName ) {
155 x = strcmp(fieldName(), e.fieldName());
156 if ( x != 0 )
157 return x;
158 }
159 x = compareElementValues(*this, e);
160 return x;
161 }

188 inline bool BSONObj::equal(const BSONObj &rhs) const {
189 BSONObjIterator i(*this);
190 BSONObjIterator j(rhs);
191 BSONElement l,r;
192 do {
193 // so far, equal...
194 l = i.next();
195 r = j.next();
196 if ( l.eoo() )
197 return r.eoo();
198 } while( l == r );
199 return false;

448 inline std::string BSONObj::toString( bool isArray, bool full ) const {
449 if ( isEmpty() ) return (isArray ? "[]" : "{}");
450 StringBuilder s;
451 toString(s, isArray, full);
452 return s.str();

454 inline void BSONObj::toString( StringBuilder& s, bool isArray, bool full, int depth ) const {
455 if ( isEmpty() ) {
456 s << (isArray ? "[]" : "{}");
457 return;
458 }
459
460 s << ( isArray ? "[ " : "{ " );
461 BSONObjIterator i(*this);
462 bool first = true;
463 while ( 1 ) {
464 massert( 10327 , "Object does not end with EOO", i.moreWithEOO() );
465 BSONElement e = i.next( true );
466 massert( 10328 , "Invalid element size", e.size() > 0 );
467 massert( 10329 , "Element too large", e.size() < ( 1 << 30 ) );
468 int offset = (int) (e.rawdata() - this->objdata());
469 massert( 10330 , "Element extends past end of object",
470 e.size() + offset <= this->objsize() );
471 bool end = ( e.size() + offset == this->objsize() );
472 if ( e.eoo() ) {
473 massert( 10331 , "EOO Before end of object", end );
474 break;
475 }
476 if ( first )
477 first = false;
478 else
479 s << ", ";
480 e.toString( s, !isArray, full, depth );
481 }
482 s << ( isArray ? " ]" : " }" );
483 }

485 inline int BSONElement::size( int maxLen ) const {
486 if ( totalSize >= 0 )
487 return totalSize;
488
489 int remain = maxLen - fieldNameSize() - 1;
490
491 int x = 0;
492 switch ( type() ) {
493 case EOO:
494 case Undefined:
495 case jstNULL:
496 case MaxKey:
497 case MinKey:
498 break;
499 case mongo::Bool:
500 x = 1;
501 break;
502 case NumberInt:
503 x = 4;
504 break;
505 case Timestamp:
506 case mongo::Date:
507 case NumberDouble:
508 case NumberLong:
509 x = 8;
510 break;
511 case jstOID:
512 x = 12;
513 break;
514 case Symbol:
515 case Code:
516 case mongo::String:
517 massert( 10313 , "Insufficient bytes to calculate element size", maxLen == -1 || remain > 3 );
518 x = valuestrsize() + 4;
519 break;
520 case CodeWScope:
521 massert( 10314 , "Insufficient bytes to calculate element size", maxLen == -1 || remain > 3 );
522 x = objsize();
523 break;
524
525 case DBRef:
526 massert( 10315 , "Insufficient bytes to calculate element size", maxLen == -1 || remain > 3 );
527 x = valuestrsize() + 4 + 12;
528 break;
529 case Object:
530 case mongo::Array:
531 massert( 10316 , "Insufficient bytes to calculate element size", maxLen == -1 || remain > 3 );
532 x = objsize();
533 break;
534 case BinData:
535 massert( 10317 , "Insufficient bytes to calculate element size", maxLen == -1 || remain > 3 );
536 x = valuestrsize() + 4 + 1/*subtype*/;
537 break;
538 case RegEx: {
539 const char *p = value();
540 size_t len1 = ( maxLen == -1 ) ? strlen( p ) : (size_t)mongo::strnlen( p, remain );
541 //massert( 10318 , "Invalid regex string", len1 != -1 ); // ERH - 4/28/10 - don't think this does anything
542 p = p + len1 + 1;
543 size_t len2;
544 if( maxLen == -1 )
545 len2 = strlen( p );
546 else {
547 size_t x = remain - len1 - 1;
548 verify( x <= 0x7fffffff );
549 len2 = mongo::strnlen( p, (int) x );
550 }
551 //massert( 10319 , "Invalid regex options string", len2 != -1 ); // ERH - 4/28/10 - don't think this does anything
552 x = (int) (len1 + 1 + len2 + 1);
553 }
554 break;
555 default: {
556 StringBuilder ss;
557 ss << "BSONElement: bad type " << (int) type();
558 std::string msg = ss.str();
559 massert( 13655 , msg.c_str(),false);
560 }
561 }
562 totalSize = x + fieldNameSize() + 1; // BSONType
563
564 return totalSize;
565 }

638 inline void BSONElement::toString( StringBuilder& s, bool includeFieldName, bool full, int depth ) const {
639
640 if ( depth > BSONObj::maxToStringRecursionDepth ) {
641 // check if we want the full/complete string
642 if ( full ) {
643 StringBuilder s;
644 s << "Reached maximum recursion depth of ";
645 s << BSONObj::maxToStringRecursionDepth;
646 uassert(16150, s.str(), full != true);
647 }
648 s << "...";
649 return;
650 }
651
652 if ( includeFieldName && type() != EOO )
653 s << fieldName() << ": ";
654 switch ( type() ) {
655 case EOO:
656 s << "EOO";
657 break;
658 case mongo::Date:
659 s << "new Date(" << (long long) date() << ')';
660 break;
661 case RegEx: {
662 s << "/" << regex() << '/';
663 const char *p = regexFlags();
664 if ( p ) s << p;
665 }
666 break;
667 case NumberDouble:
668 s.appendDoubleNice( number() );
669 break;
670 case NumberLong:
671 s << _numberLong();
672 break;
673 case NumberInt:
674 s << _numberInt();
675 break;
676 case mongo::Bool:
677 s << ( boolean() ? "true" : "false" );
678 break;
679 case Object:
680 embeddedObject().toString(s, false, full, depth+1);
681 break;
682 case mongo::Array:
683 embeddedObject().toString(s, true, full, depth+1);
684 break;
685 case Undefined:
686 s << "undefined";
687 break;
688 case jstNULL:
689 s << "null";
690 break;
691 case MaxKey:
692 s << "MaxKey";
693 break;
694 case MinKey:
695 s << "MinKey";
696 break;
697 case CodeWScope:
698 s << "CodeWScope( "
699 << codeWScopeCode() << ", " << codeWScopeObject().toString(false, full) << ")";
700 break;
701 case Code:
702 if ( !full && valuestrsize() > 80 ) {
703 s.write(valuestr(), 70);
704 s << "...";
705 }
706 else {
707 s.write(valuestr(), valuestrsize()-1);
708 }
709 break;
710 case Symbol:
711 case mongo::String:
712 s << '"';
713 if ( !full && valuestrsize() > 160 ) {
714 s.write(valuestr(), 150);
715 s << "...\"";
716 }
717 else {
718 s.write(valuestr(), valuestrsize()-1);
719 s << '"';
720 }
721 break;
722 case DBRef:
723 s << "DBRef('" << valuestr() << "',";
724 {
725 mongo::OID *x = (mongo::OID *) (valuestr() + valuestrsize());
726 s << *x << ')';
727 }
728 break;
729 case jstOID:
730 s << "ObjectId('";
731 s << __oid() << "')";
732 break;
733 case BinData:
734 s << "BinData(" << binDataType() << ", ";
735 {
736 int len;
737 const char *data = binDataClean(len);
738 if ( !full && len > 80 ) {
739 s << toHex(data, 70) << "...)";
740 }
741 else {
742 s << toHex(data, len) << ")";
743 }
744 }
745 break;
746 case Timestamp:
747 s << "Timestamp " << timestampTime() << "|" << timestampInc();
748 break;
749 default:
750 s << "?type=" << type();
751 break;
752 }
753 }

758 inline BSONElement BSONObj::getFieldDotted(const StringData& name) const {
759 BSONElement e = getField(name);
760 if (e.eoo()) {
761 size_t dot_offset = name.find('.');
762 if (dot_offset != std::string::npos) {
763 StringData left = name.substr(0, dot_offset);
764 StringData right = name.substr(dot_offset + 1);
765 BSONObj sub = getObjectField(left);
766 return sub.isEmpty() ? BSONElement() : sub.getFieldDotted(right);
767 }
768 }
769
770 return e;
771 }

791 inline BSONObj::BSONObj() {
792 /* little endian ordering here, but perhaps that is ok regardless as BSON is spec'd
793 to be little endian external to the system. (i.e. the rest of the implementation of bson,

891 inline std::string escape( const std::string& s , bool escape_slash=false) {
892 StringBuilder ret;
893 for ( std::string::const_iterator i = s.begin(); i != s.end(); ++i ) {
894 switch ( *i ) {
895 case '"':
896 ret << "\\\"";
897 break;
898 case '\\':
899 ret << "\\\\";
900 break;
901 case '/':
902 ret << (escape_slash ? "\\/" : "/");
903 break;
904 case '\b':
905 ret << "\\b";
906 break;
907 case '\f':
908 ret << "\\f";
909 break;
910 case '\n':
911 ret << "\\n";
912 break;
913 case '\r':
914 ret << "\\r";
915 break;
916 case '\t':
917 ret << "\\t";
918 break;
919 default:
920 if ( *i >= 0 && *i <= 0x1f ) {
921 //TODO: these should be utf16 code-units not bytes
922 char c = *i;
923 ret << "\\u00" << toHexLower(&c, 1);
924 }
925 else {
926 ret << *i;
927 }
928 }
929 }
930 return ret.str();
931 }

933 inline std::string BSONObj::hexDump() const {
934 std::stringstream ss;
935 const char *d = objdata();
936 int size = objsize();
937 for( int i = 0; i < size; ++i ) {
938 ss.width( 2 );
939 ss.fill( '0' );
940 ss << std::hex << (unsigned)(unsigned char)( d[ i ] ) << std::dec;
941 if ( ( d[ i ] >= '0' && d[ i ] <= '9' ) || ( d[ i ] >= 'A' && d[ i ] <= 'z' ) )
942 ss << '\'' << d[ i ] << '\'';
943 if ( i != size - 1 )
944 ss << ' ';
945 }
946 return ss.str();
947 }
