385// sfinae in the bool return to avoid competing with the value == value
386// operators
387template <typename T>
388using not_value = typename std::enable_if<
389 std::is_constructible<value, T>::value &&
390 !std::is_same<typename std::remove_reference<T>::type, value>::value,
391 bool>::type;
392
393// these all return bool
394template <typename T>
395BSONCXX_INLINE not_value<T> operator==(const value& lhs, T&& rhs) {
396 return lhs == value{std::forward<T>(rhs)};
397}
398
399template <typename T>
400BSONCXX_INLINE not_value<T> operator==(T&& lhs, const value& rhs) {
401 return value{std::forward<T>(lhs)} == rhs;
402}
403
404template <typename T>
405BSONCXX_INLINE not_value<T> operator!=(const value& lhs, T&& rhs) {
406 return lhs != value{std::forward<T>(rhs)};
407}
408
409template <typename T>
410BSONCXX_INLINE not_value<T> operator!=(T&& lhs, const value& rhs) {
411 return value{std::forward<T>(lhs)} != rhs;
412}
