57 inline std::string BSONElement::_asCode() const {
58 switch( type() ) {
59 case mongo::String:
60 case Code:
61 return std::string(valuestr(), valuestrsize()-1);
62 case CodeWScope:
63 return std::string(codeWScopeCode(), *(int*)(valuestr())-1);
64 default:
65 log() << "can't convert type: " << (int)(type()) << " to code" << std::endl;
66 }
67 uassert( 10062 , "not code" , 0 );
68 return "";
69 }
