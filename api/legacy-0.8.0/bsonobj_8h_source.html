502#pragma pack(1)
503 class Holder : boost::noncopyable {
504 private:
505 Holder(); // this class should never be explicitly created
506 AtomicUInt refCount;
507 public:
508 char data[4]; // start of object
509
510 void zero() { refCount.zero(); }
511
512 // these are called automatically by boost::intrusive_ptr
513 friend void intrusive_ptr_add_ref(Holder* h) { h->refCount++; }
514 friend void intrusive_ptr_release(Holder* h) {
515#if defined(_DEBUG) // cant use dassert or DEV here
516 verify((int)h->refCount > 0); // make sure we haven't already freed the buffer
517#endif
518 if(--(h->refCount) == 0){
519#if defined(_DEBUG)
520 unsigned sz = (unsigned&) *h->data;
521 verify(sz < BSONObjMaxInternalSize * 3);
522 memset(h->data, 0xdd, sz);
523#endif
524 free(h);
525 }
526 }
