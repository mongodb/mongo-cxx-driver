678  char* _done() {
679  if ( _doneCalled )
680  return _b.buf() + _offset;
681 
682  _doneCalled = true;
683  _s.endField();
684  _b.appendNum((char) EOO);
685  char *data = _b.buf() + _offset;

687  *((int*)data) = size;
688  if ( _tracker )
689  _tracker->got( size );
690  return data;
691  }
