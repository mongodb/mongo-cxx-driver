678 char* _done() {
679 if ( _doneCalled )
680 return _b.buf() + _offset;
681
682 _doneCalled = true;
683 _s.endField();
684 _b.appendNum((char) EOO);
685 char *data = _b.buf() + _offset;
686 int size = _b.len() - _offset;
687 *((int*)data) = size;
688 if ( _tracker )
689 _tracker->got( size );
690 return data;
691 }

858 void fill (int upTo){
859 // if this is changed make sure to update error message and jstests/set7.js
860 const int maxElems = 1500000;
861 BOOST_STATIC_ASSERT(maxElems < (BSONObjMaxUserSize/10));
862 uassert(15891, "can't backfill array to larger than 1,500,000 elements", upTo <= maxElems);
863
864 while( _i < upTo )
865 appendNull();
866 }
