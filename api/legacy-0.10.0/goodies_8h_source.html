134 class ThreadSafeString : boost::noncopyable {
135 public:
136 ThreadSafeString( size_t size=256 )
137 : _size( size ) , _buf( new char[size] ) {
138 memset( _buf , 0 , _size );
139 }
140
141 ~ThreadSafeString() {
142 delete[] _buf;
143 }
144
145 std::string toString() const {
146 std::string s = _buf;
147 return s;
148 }
149
150 ThreadSafeString& operator=( const char * str ) {
151 size_t s = strlen(str);
152 if ( s >= _size - 2 )
153 s = _size - 2;
154 strncpy( _buf , str , s );
155 _buf[s] = 0;
156 return *this;
157 }
158
159 bool empty() const {
160 return _buf == 0 || _buf[0] == 0;
161 }
162
163 private:
164 const size_t _size;
165 char *const _buf;
166 };
