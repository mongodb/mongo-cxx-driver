63 BSONElement next( bool checkEnd ) {
64 verify( _pos <= _theend );
65
66 int maxLen = -1;
67 if ( checkEnd ) {
68 maxLen = _theend + 1 - _pos;
69 verify( maxLen > 0 );
70 }
71
72 BSONElement e( _pos, maxLen );
73 int esize = e.size( maxLen );
74 massert( 16446, "BSONElement has bad size", esize > 0 );
75 _pos += esize;
76
77 return e;
78 }

83 return e;
84 }
