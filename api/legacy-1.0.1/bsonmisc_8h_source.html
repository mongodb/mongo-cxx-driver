240 class BSONSizeTracker {
241 public:
242 BSONSizeTracker() {
243 _pos = 0;
244 for ( int i=0; i<SIZE; i++ )
245 _sizes[i] = 512; // this is the default, so just be consistent
246 }
247
248 ~BSONSizeTracker() {
249 }
250
251 void got( int size ) {
252 _sizes[_pos] = size;
253 _pos = (_pos + 1) % SIZE; // thread safe at least on certain compilers
254 }
255
259 int getSize() const {
260 int x = 16; // sane min
261 for ( int i=0; i<SIZE; i++ ) {
262 if ( _sizes[i] > x )
263 x = _sizes[i];
264 }
265 return x;
266 }
267
268 private:
269 enum { SIZE = 10 };
270 int _pos;
271 int _sizes[SIZE];
272 };
